Name Based Rep
--------------

example:

<hc id="system">
  <attachments>
    <att mandatory="0"
    	 id="/system/agroup_authz_insts/agroup_authz_inst:global default deny"/>
    <att mandatory="1"
    	 id="/system/agroup_authz_insts/agroup_authz_inst:global admins allow"/>
  </attachments>
  <hc id="sets">
    <obj id="set:global admins">
      <field id="member_type">2</field>
      <members>
        <ref id="/path/to/hc/type:obj"/>
	<ref id="2:1"/>
      </members>
    </obj>
  </hc>
</hc>
